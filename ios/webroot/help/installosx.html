translate("Install Bibledit-Web 1.9 on OSX")

translate("This assumes a standard installation of the operating system.") translate("Bibledit-Web runs on all recent versions of OSX.") translate("The steps below have been tested on OS X 10.7 Lion.") translate("Open a terminal to enter some of the commands:") Applications / Utilities / Terminal.

  1. translate("Install Xcode.")

  2. translate("Install MacPorts.")

  3. translate("Install the Apache web server:")

    sudo port install apache2

  4. translate("Activate Apache:")

    sudo port load apache2

  5. translate("To verify Apache is running, point your browser to:") ​http://localhost. translate("The page should say:") It works!

  6. translate("Install PHP plus modules:")

    sudo port install php5 +apache2

    sudo port install php5-mbstring

    sudo port install php5-posix

    sudo port install php5-sqlite

    sudo port install php5-iconv

  7. translate("Register PHP with Apache:")

    cd /opt/local/apache2/modules

    sudo /opt/local/apache2/bin/apxs -a -e -n "php5" libphp5.so

  8. translate("Update the Apache configuration:")

    sudo nano /opt/local/apache2/conf/httpd.conf

    translate("In that file, search for:")

    DirectoryIndex index.html

    translate("and change it to:")

    DirectoryIndex index index.html

    translate("Add the following line at the end of that file:")

    Include conf/extra/mod_php.conf

    translate("Save the file.")

  9. translate("Configure PHP:")

    cd /opt/local/etc/php5

    sudo cp php.ini-development php.ini

  10. translate("Restart Apache for the new configuration to take effect:")

    sudo /opt/local/apache2/bin/apachectl restart

  11. translate("Download") Bibledit-Web 1.9.

  12. translate("Unzip Bibledit-Web into this folder:") /opt/local/apache2/htdocs

  13. translate("Set correct permissions on folder bibledit-web for the web server:")

    sudo chown -R root /opt/local/apache2/htdocs/bibledit-web

    sudo chmod -R 0777 /opt/local/apache2/htdocs/bibledit-web

  14. translate("To configure Bibledit-Web, point your browser to:") ​http://localhost/bibledit-web. translate("Follow all the steps in the browser.")

  15. translate("To use Bibledit-Web, point your browser to the same address:") ​http://localhost/bibledit-web.

  16. translate("Once everything works, put PHP in production mode:")

    cd /opt/local/etc/php5

    sudo cp php.ini-production php.ini
