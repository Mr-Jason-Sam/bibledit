gettext("Install Bibledit-Web 1.9 on Ubuntu 14.04")

gettext("Follow the steps below to install Bibledit-Web.") gettext("It needs some basic knowledge of working with Linux servers.")

  1. gettext("Open a terminal.")

  2. gettext("Update the software sources:")

    $ sudo apt-get update

  3. gettext("Start installing the web server:")

    $ sudo apt-get install tasksel

    $ sudo tasksel

  4. gettext("Enable the LAMP server, and install.")
  5. gettext("Test the web server by opening the site in the browser, for example:") http://localhost gettext("or") http://website.org. gettext("When the web server works, the browser displays the default Apache page.")

  6. gettext("Remove the MySQL server to save memory:")

    $ sudo apt-get remove --purge mysql-server* mysql-common

  7. gettext("Install the software Bibledit-Web relies on:")

    $ sudo apt-get install git unzip php5-json php5-sqlite

  8. gettext("Reload the Apache web server so that the new modules take effect:")

    $ sudo service apache2 reload

  9. gettext("Download Bibledit-Web:")

    $ sudo wget http://download.savannah.nongnu.org/releases/bibledit-web/source/bibledit-web-1.9.zip

  10. gettext("Install Bibledit-Web in folder /var/www/html as user root:")

    $ cd /var/www/html

    $ sudo unzip ~/bibledit-web-1.9.zip

  11. gettext("Make everything to be owned by the web server user:")

    $ sudo chown -R www-data:www-data bibledit-web

  12. gettext("Bibledit-Web will be accessible through:") http://website.org/bibledit-web.
  13. gettext("Open the web address in the browser, and follow the steps on the screen to configure Bibledit-Web and log in.")
  14. gettext("Enable the use of .htaccess files in Apache to secure some Bibledit-Web folders.") gettext("Edit file apache2.conf instead of the one mentioned in the guide.")
