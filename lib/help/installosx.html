gettext("Install Bibledit-Web 1.9 on OSX")

gettext("This assumes a standard installation of the operating system.") gettext("Bibledit-Web runs on all recent versions of OSX.") gettext("The steps below have been tested on OS X 10.7 Lion.") gettext("Open a terminal to enter some of the commands:") Applications / Utilities / Terminal.

  1. gettext("Install Xcode.")

  2. gettext("Install MacPorts.")

  3. gettext("Install the Apache web server:")

    sudo port install apache2

  4. gettext("Activate Apache:")

    sudo port load apache2

  5. gettext("To verify Apache is running, point your browser to:") ​http://localhost. gettext("The page should say:") It works!

  6. gettext("Install PHP plus modules:")

    sudo port install php5 +apache2

    sudo port install php5-mbstring

    sudo port install php5-posix

    sudo port install php5-sqlite

    sudo port install php5-iconv

  7. gettext("Register PHP with Apache:")

    cd /opt/local/apache2/modules

    sudo /opt/local/apache2/bin/apxs -a -e -n "php5" libphp5.so

  8. gettext("Update the Apache configuration:")

    sudo nano /opt/local/apache2/conf/httpd.conf

    gettext("In that file, search for:")

    DirectoryIndex index.html

    gettext("and change it to:")

    DirectoryIndex index.php index.html

    gettext("Add the following line at the end of that file:")

    Include conf/extra/mod_php.conf

    gettext("Save the file.")

  9. gettext("Configure PHP:")

    cd /opt/local/etc/php5

    sudo cp php.ini-development php.ini

  10. gettext("Restart Apache for the new configuration to take effect:")

    sudo /opt/local/apache2/bin/apachectl restart

  11. gettext("Download") Bibledit-Web 1.9.

  12. gettext("Unzip Bibledit-Web into this folder:") /opt/local/apache2/htdocs

  13. gettext("Set correct permissions on folder bibledit-web for the web server:")

    sudo chown -R root /opt/local/apache2/htdocs/bibledit-web

    sudo chmod -R 0777 /opt/local/apache2/htdocs/bibledit-web

  14. gettext("To configure Bibledit-Web, point your browser to:") ​http://localhost/bibledit-web. gettext("Follow all the steps in the browser.")

  15. gettext("To use Bibledit-Web, point your browser to the same address:") ​http://localhost/bibledit-web.

  16. gettext("Once everything works, put PHP in production mode:")

    cd /opt/local/etc/php5

    sudo cp php.ini-production php.ini
